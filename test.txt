model ScheduleReplacement {
  id         String   @id @default(uuid())
  date       DateTime
  scheduleId String
  teacherId  String // pengganti

  schedule   Schedule @relation(fields: [scheduleId], references: [id])
  teacher    Teacher  @relation(fields: [teacherId], references: [id])

  createdAt  DateTime @default(now())
  updatedAt  DateTime @updatedAt

  @@unique([scheduleId, date])
}


import { prisma } from './prisma'; // Sesuaikan dengan setupmu
import { startOfDay } from 'date-fns';

async function getTodayClassStatus(today: Date) {
  const dayOfWeek = today.getDay() === 0 ? 7 : today.getDay(); // 1-7

  const schedules = await prisma.schedule.findMany({
    where: { dayOfWeek },
    include: {
      teacher: true,
      subject: true,
      class: true,
      scheduleSlot: true,
    },
  });

  const absences = await prisma.teacherScheduleAbsence.findMany({
    where: { date: startOfDay(today) },
  });

  const replacements = await prisma.scheduleReplacement.findMany({
    where: { date: startOfDay(today) },
    include: { teacher: true },
  });

  const result = schedules.map(schedule => {
    const absence = absences.find(
      a => a.scheduleId === schedule.id && a.teacherId === schedule.teacherId,
    );

    const replacement = replacements.find(
      r => r.scheduleId === schedule.id,
    );

    let status = 'BELUM_DIISI';
    let replacementName = null;

    if (absence?.status === 'PRESENT') {
      status = 'HADIR';
    } else if (replacement) {
      status = 'DIGANTIKAN';
      replacementName = replacement.teacher.name;
    } else if (absence) {
      status = 'KOSONG';
    }

    return {
      class: schedule.class.name,
      slot: schedule.scheduleSlot.slot,
      time: `${schedule.scheduleSlot.startTime} - ${schedule.scheduleSlot.endTime}`,
      subject: schedule.subject.name,
      teacher: schedule.teacher.name,
      status,
      replacement: replacementName,
    };
  });

  return result;
}
